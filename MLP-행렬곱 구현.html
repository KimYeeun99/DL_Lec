4. sigmoid, softmax 함수 구현¶

$ g(z) = \frac{1}{1 + e^{-z}} $¶

In [5]:
def sigmoid(x):
    return 1/(1+np.exp(-x))

$ softmax(x)_i = \frac{e^{x_i}} {\Sigma_{j} e^{x_j}} $¶

In [6]:
def softmax(x):
    e_x=np.exp(x)
    return e_x/np.sum(e_x)

5. 모델 선언¶

# Multi-layered perceptron
# # of units in each layer : 28*28 - 258 -128 - 10
class MyMLP:
    def __init__(self, n_input, n_hidden1, n_hidden2, n_output):
        # W^(1): layer1->layer2에 매핑되는 Weight
        self.W1 = np.zeros((n_hidden1, n_input), dtype=np.float32)
        self.b1 = np.zeros((n_hidden1,), dtype=np.float32)
        
        self.W2 = np.zeros((n_hidden2, n_hidden1), dtype=np.float32)
        self.b2 = np.zeros((n_hidden2,), dtype=np.float32)
        
        self.W3 = np.zeros((n_output, n_hidden2), dtype=np.float32)
        self.b3 = np.zeros((n_output), dtype=np.float32)
        
    def __call__(self, x):
        # (1,28,28) -> (28*28)
        x = x.reshape(-1) # 일렬로 피기
        
        h1 = sigmoid(np.dot(self.W1, x) + self.b1) # W1(258, 28*28), x(28*28), b1(256) -> h1(256)
        h2 = np.dot(self.W2, h1) +self.b2 # W2(128, 256), h1(256), b2(128) ->  h2(128)
        out = np.dot(self.W3, h2) +self.b3 # W3(10, 128), h2(128), b3(10) -> out(10)
        
        return softmax(out) # (10)
